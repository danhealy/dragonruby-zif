def float_to(obj)
  run_action(
    new_action(
      {x: :center_x, y: :center_y},
      follow: obj,
      duration: random_duration,
      easing: :linear
    ) do
      # Repeat with new durations and color.
      @x = 1659
      @y = 1659
      randomize_color
      @spin_action.duration = random_duration
      float_to(obj)
    end
  )
end

def spin
  @spin_action = new_action(
    {angle: 359},
    duration: random_duration,
    easing: :linear,
    repeat: :forever
  )
  run_action(@spin_action)
end
